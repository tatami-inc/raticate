167 static std::pair<Index_, Index_> round_indices(Index_ i, Index_ interval, Index_ max) {
168 Index_ new_first = (i / interval) * interval;
169 Index_ new_last = std::min(max, new_first + interval);
170 return std::make_pair(new_first, new_last - new_first);
171 }
172
173 static Rcpp::IntegerVector create_consecutive_indices(Index_ start, Index_ length) {
174 Rcpp::IntegerVector out(length);
175 std::iota(out.begin(), out.end(), start + 1); // 1-based
176 return out;
177 }

179 template<bool byrow_, bool sparse_>
180 Rcpp::List create_rounded_indices(Index_ i, Workspace<sparse_>* work) const {
181 Rcpp::List indices(2);
182 if constexpr(byrow_) {
183 auto row_rounded = round_indices(i, block_nrow, internal_nrow);
184 work->primary_block_start = row_rounded.first;
185 work->primary_block_len = row_rounded.second;
186 indices[0] = create_consecutive_indices(row_rounded.first, row_rounded.second);
187 indices[1] = work->secondary_indices;
188 } else {
189 auto col_rounded = round_indices(i, block_ncol, internal_ncol);
190 work->primary_block_start = col_rounded.first;
191 work->primary_block_len = col_rounded.second;
192 indices[0] = work->secondary_indices;
193 indices[1] = create_consecutive_indices(col_rounded.first, col_rounded.second);
194 }
195 return indices;
196 }

198 template<bool byrow_, bool sparse_, bool sparse_err = sparse_>
199 void check_buffered_dims(const tatami::Matrix<Value_, Index_>* parsed, const Workspace<sparse_>* work) const {
200 size_t parsed_primary = (byrow_ ? parsed->nrow() : parsed->ncol());
201 size_t parsed_secondary = (byrow_ ? parsed->ncol() : parsed->nrow());
202
203 if (parsed_primary != work->primary_block_len || parsed_secondary != work->secondary_len) {
204 auto ctype = get_class_name(original_seed);
205 throw std::runtime_error("'" +
206 (sparse_err ? std::string("extract_sparse_array") : std::string("extract_array")) +
207 "(<" + ctype + ">)' returns incorrect dimensions");
208 }
209 }

354 tatami::SparseRange<Value_, Index_> fetch(Index_ i, Value_* vbuffer, Index_* ibuffer) {
355 if (this->parent->needs_reset(i, this->work.get())) {
356#ifdef TATAMI_R_PARALLELIZE_UNKNOWN
357 auto& mexec = executor();
358 mexec.run([&]() -> void {
359#endif
360 this->parent->template run_sparse_extractor<byrow>(i, options, this->work.get());
361#ifdef TATAMI_R_PARALLELIZE_UNKNOWN
362 });
363#endif
364 }
365
366 i -= this->work->primary_block_start;
367 tatami::SparseRange<Value_, Index_> output = this->work->bufextractor->fetch_copy(i, vbuffer, ibuffer);
368
369 // Need to adjust the indices.
370 if (output.index) {
371 if constexpr(selection_ == tatami::DimensionSelectionType::BLOCK) {
372 for (size_t i = 0; i < output.number; ++i) {
373 ibuffer[i] = output.index[i] + this->block_start;
374 }
375 output.index = ibuffer;
376 } else if constexpr(selection_ == tatami::DimensionSelectionType::INDEX) {
377 for (size_t i = 0; i < output.number; ++i) {
378 ibuffer[i] = this->indices[output.index[i]];
379 }
380 output.index = ibuffer;
381 }
382 }
383
384 return output;
385 }
